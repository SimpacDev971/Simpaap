    <script>
        pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.worker.min.js';

        let pdfDoc = null;
        let pageNum = 1;
        let scale = 1.5; // High resolution preview
        const canvas = document.getElementById('pdf-canvas');
        const ctx = canvas.getContext('2d');
        const overlay = document.getElementById('window-overlay');
        const container = document.getElementById('pdf-container');

        // Conversion factor: 1mm = 72/25.4 points
        const MM_TO_PT = 72 / 25.4;

        async function renderPage(num) {
            const page = await pdfDoc.getPage(num);
            const viewport = page.getViewport({ scale: scale });
            
            canvas.height = viewport.height;
            canvas.width = viewport.width;

            const renderContext = {
                canvasContext: ctx,
                viewport: viewport
            };

            await page.render(renderContext).promise;
            updateOverlay(viewport);
            
            document.getElementById('page-info').textContent = `Page ${num} of ${pdfDoc.numPages}`;
            document.getElementById('viewer-wrapper').classList.remove('hidden');
            document.getElementById('placeholder').classList.add('hidden');
        }

        function updateOverlay(viewport) {
            const xMm = parseFloat(document.getElementById('win-x').value);
            const yMm = parseFloat(document.getElementById('win-y').value);
            const wMm = parseFloat(document.getElementById('win-w').value);
            const hMm = parseFloat(document.getElementById('win-h').value);

            // PDF.js uses scale to render. We must apply that scale to our mm->pt points
            const xPx = xMm * MM_TO_PT * scale;
            const yPx = yMm * MM_TO_PT * scale;
            const wPx = wMm * MM_TO_PT * scale;
            const hPx = hMm * MM_TO_PT * scale;

            overlay.style.left = `${xPx}px`;
            overlay.style.top = `${yPx}px`;
            overlay.style.width = `${wPx}px`;
            overlay.style.height = `${hPx}px`;
            overlay.style.display = 'block';
        }

        document.getElementById('file-input').addEventListener('change', async (e) => {
            const file = e.target.files[0];
            if (file && file.type === 'application/pdf') {
                const reader = new FileReader();
                reader.onload = async function() {
                    const typedarray = new Uint8Array(this.result);
                    pdfDoc = await pdfjsLib.getDocument(typedarray).promise;
                    pageNum = 1;
                    renderPage(pageNum);
                };
                reader.readAsArrayBuffer(file);
            }
        });

        document.getElementById('apply-btn').addEventListener('click', () => {
            if (pdfDoc) {
                // Re-render to ensure viewport math is fresh
                renderPage(pageNum);
            }
        });

        document.getElementById('prev-page').addEventListener('click', () => {
            if (pageNum <= 1) return;
            pageNum--;
            renderPage(pageNum);
        });

        document.getElementById('next-page').addEventListener('click', () => {
            if (pageNum >= pdfDoc.numPages) return;
            pageNum++;
            renderPage(pageNum);
        });

        // Add keyboard support for nudging
        window.addEventListener('keydown', (e) => {
            if (!pdfDoc) return;
            const step = e.shiftKey ? 5 : 1;
            const inputs = {
                'ArrowUp': 'win-y',
                'ArrowDown': 'win-y',
                'ArrowLeft': 'win-x',
                'ArrowRight': 'win-x'
            };

            if (inputs[e.key]) {
                e.preventDefault();
                const el = document.getElementById(inputs[e.key]);
                let val = parseFloat(el.value);
                if (e.key === 'ArrowUp' || e.key === 'ArrowLeft') val -= step;
                else val += step;
                el.value = val;
                renderPage(pageNum);
            }
        });
    </script>
